[٣] عين قيمة جـ التي تجعل المقدار مربعاً كاملاً :-
(١) \( ١٦س^٢ + جـ س ص + ٤٩ص^٢ \) (٢) \( جـ س^٢ + ٢٨س + ٤٩ \)
(٣) \( ٩س^٢ - ١٢س ص + جـ \)
* الحل * نعلم أن الحد الأوسط =
\( \pm ٢ \times \sqrt{\text{الحد الأول}} \times \sqrt{\text{الجذر التربيعي للحد الثالث}} \)
(١) \( جـ ص ص = \pm ٢ \times ٤س \times ٧ص \) بالقسمة على \( س ص \) نجد
\( \frac{جـ س ص}{س ص} = \frac{\pm ٢ \times ٤س \times ٧ص}{س ص} \) يكون \( جـ = \pm ٢ \times ٤ \times ٧ = \pm ٥٦ \)
(٢) \( \because ٢٨س = \pm ٢ \times \sqrt{جـ}س \times ٧ \) بالقسمة على س
\( ٢٨ = \pm ٢ \times \sqrt{جـ} \times ٧ \)
\( \frac{٢٨}{١٤} = \frac{\pm ١٤ \sqrt{جـ}}{١٤} \) يكون \( ٢٨ = ١٤\sqrt{جـ} \) بالقسمة على ١٤
\( \sqrt{جـ} = ٢ \) بالتربيع يكون \( جـ = ٤ \)
(٣) \( \because -١٢س ص = -٢ \times ٣س \times \sqrt{جـ} \) بالقسمة على -٦س
\( \frac{-١٢س ص}{-٦س} = \frac{-٦س \sqrt{جـ}}{-٦س} \) فإن \( \sqrt{جـ} = ٢ص \)
بالتربيع \( جـ = ٤ص^٢ \)

[٤] حل ما يأتي :-
(١) \( س^٤ + ٩ص^٤ + ٨١ص^٤ \) بإضافة وطرح الحد الأوسط
الحد الأوسط = \( \pm ٢ \times س^٢ \times ٩ص^٢ = \pm ١٨س^٢ ص^٢ \neq ٩س^٢ ص^٢ \)
\( \therefore س^٤ + ٩س^٢ ص^٢ + ٨١ص^٤ \)
\( = س^٤ + ١٨س^٢ ص^٢ + ٨١ص^٤ + ٩س^٢ ص^٢ - ١٨س^٢ ص^٢ \)
مربع كامل
المقدار = \( (س^٢ + ٩ص^٢)^٢ - ٩س^٢ ص^٢ \) كفرق بين مربعين
\( = [(س^٢ + ٩ص^٢) + ٣س ص] [(س^٢ + ٩ص^٢) - ٣س ص] \)
\( = (س^٢ + ٣س ص + ٩ص^٢) (س^٢ - ٣س ص + ٩ص^٢) \)
(٢) \( ١٦س^٤ + ٢٤س^٢ ص^٢ + ٢٥ص^٤ \)
الحد الأوسط = \( \pm ٢ \times ٤س^٢ \times ٥ص^٢ = \pm ٤٠س^٢ ص^٢ \neq ٢٤س^٢ ص^٢ \)
المقدار = \( ١٦س^٤ + ٢٤س^٢ ص^٢ + ٢٥ص^٤ \)
\( = ١٦س^٤ + ٤٠س^٢ ص^٢ + ٢٥ص^٤ - ٤٠س^٢ ص^٢ + ٢٤س^٢ ص^٢ \)
\( = ١٦س^٤ + ٤٠س^٢ ص^٢ + ٢٥ص^٤ - ١٦س^٢ ص^٢ \)
\( = (٤س^٢ + ٥ص^٢)^٢ - ١٦س^٢ ص^٢ \)
\( = [(٤س^٢ + ٥ص^٢) + ٤س ص] [(٤س^٢ + ٥ص^٢) - ٤س ص] \)
\( = (٤س^٢ + ٤س ص + ٥ص^٢) (٤س^٢ - ٤س ص + ٥ص^٢) \)
(٣) \( ٨ل^٤ - ٥٠ل^٢ م^٢ + ٧٢م^٤ \)
واضح أن الحد الثالث ليس مربعاً كاملاً بأخذ مشترك ٢ يكون المقدار
\( = ٢ ( ٤ل^٤ - ٢٥ل^٢ م^٢ + ٣٦م^٤ ) \) نضيف ونطرح
حيث الحد الأوسط = \( \pm ٢ \times ٢ل^٢ \times ٦م^٢ = \pm ٢٤ل^٢ م^٢ \neq -٢٥ل^٢ م^٢ \)
المقدار = \( ٢ [ ٤ل^٤ - ٢٤ل^٢ م^٢ + ٣٦م^٤ - ٢٤ل^٢ م^٢ - ٢٥ل^٢ م^٢ ] \)
\( = ٢ [ (٢ل^٢ + ٦م^٢)^٢ - ٤٩ل^٢ م^٢ ] \)
\( = ٢ [ (٢ل^٢ + ٦م^٢) + ٧ل م ] [ (٢ل^٢ + ٦م^٢) - ٧ل م ] \)
\( = ٢ (٢ل^٢ + ٧ل م + ٦م^٢) (٢ل^٢ - ٧ل م + ٦م^٢) \)
(٤) \( ١,٢س^٤ + ٧,٥ص^٤ - ٢,١س^٢ ص^٢ \) نأخذ مشترك ١,٢ مع ترتيب الحدود
المقدار = \( ١,٢ ( س^٤ - ٢س^٢ ص^٢ + \frac{٧,٥}{١,٢} ص^٤ ) \)
\( = ١,٢ ( س^٤ - ٦س^٢ ص^٢ + \frac{٢٥}{٤} ص^٤ ) \)
نجد الأوسط ثم يضاف ويطرح
الحد الأوسط = \( \pm ٢ \times س^٢ \times \frac{٥}{٢} ص^٢ = \pm ٥س^٢ ص^٢ \)
المقدار = \( ١,٢ ( س^٤ - ٥س^٢ ص^٢ + \frac{٢٥}{٤} ص^٤ + ٥س^٢ ص^٢ - ٦س^٢ ص^٢ ) \)
\( = ١,٢ [ (س^٢ - \frac{٥}{٢} ص^٢)^٢ - س^٢ ص^٢ ] \)
\( = ١,٢ [ (س^٢ - \frac{٥}{٢} ص^٢) + س ص ] [ (س^٢ - \frac{٥}{٢} ص^٢) - س ص ] \)
\( = ١,٢ (س^٢ + س ص - \frac{٥}{٢} ص^٢) (س^٢ - س ص - \frac{٥}{٢} ص^٢) \)
(٥) \( أ^٤ + ٤ب^٤ \)
الحد الأوسط = \( ٢ \times أ^٢ \times ٢ب^٢ = ٤أ^٢ ب^٢ \) يضاف الموجب والسالب
المقدار = \( أ^٤ + ٤ب^٤ = أ^٤ + ٤أ^٢ ب^٢ + ٤ب^٤ - ٤أ^٢ ب^٢ \)
\( = (أ^٢ + ٢ب^٢)^٢ - ٤أ^٢ ب^٢ \)
\( = [(أ^٢ + ٢ب^٢) + ٢أ ب] [(أ^٢ + ٢ب^٢) - ٢أ ب] \)
\( = (أ^٢ + ٢أ ب + ٢ب^٢) (أ^٢ - ٢أ ب + ٢ب^٢) \)
(٦) \( ٦٢٥س^٤ + ٤ص^٤ \)
الحد الأوسط = \( \pm ٢ \times ٢٥س^٢ \times ٢ص^٢ = \pm ١٠٠س^٢ ص^٢ \)
المقدار = \( ٦٢٥س^٤ + ١٠٠س^٢ ص^٢ + ٤ص^٤ - ١٠٠س^٢ ص^٢ \)
المقدار = \( (٢٥س^٢ + ٢ص^٢)^٢ - ١٠٠س^٢ ص^٢ \)
\( = [(٢٥س^٢ + ٢ص^٢) + ١٠س ص] [(٢٥س^٢ + ٢ص^٢) - ١٠س ص] \)
\( = (٢٥س^٢ + ١٠س ص + ٢ص^٢) (٢٥س^٢ - ١٠س ص + ٢ص^٢) \)
(٧) \( ٦٤أ^٤ + ب^٤ = ٦٤أ^٤ + ١٦أ^٢ ب^٢ + ب^٤ - ١٦أ^٢ ب^٢ \)
\( (٨أ^٢ + ب^٢)^٢ - ١٦أ^٢ ب^٢ \)
\( = [(٨أ^٢ + ب^٢) + ٤أ ب] [(٨أ^٢ + ب^٢) - ٤أ ب] \)
\( = (٨أ^٢ + ٤أ ب + ب^٢) (٨أ^٢ - ٤أ ب + ب^٢) \)
(٨) \( ص^٤ - ٧ص^٢ ع^٤ + ع^٨ = ص^٤ + ٢ص^٢ ع^٤ + ع^٨ - ٢ص^٢ ع^٤ - ٧ص^٢ ع^٤ \)
\( = (ص^٢ + ع^٤)^٢ - ٩ص^٢ ع^٤ \)
\( = [(ص^٢ + ع^٤) + ٣ص ع^٢] [(ص^٢ + ع^٤) - ٣ص ع^٢] \)
\( = (ص^٢ + ٣ص ع^٢ + ع^٤) (ص^٢ - ٣ص ع^٢ + ع^٤) \)
(٩) \( س^٤ + ١٣س^٢ + ٣٦ = (س^٤ + ١٢س^٢ + ٣٦) + س^٢ \)
\( = (س^٢ + ٦)^٢ + س^٢ \)
\( = [(س^٢ + ٦) - س] [(س^٢ + ٦) + س] \)
\( = (س^٢ - س + ٦) (س^٢ + س + ٦) \)
(١٠) \( جـ^٤ + ٣جـ^٢ د^٢ + ٤د^٤ \)
الحد الأوسط = \( \pm ٢ \times جـ^٢ \times ٢د^٢ = \pm ٤جـ^٢ د^٢ \)
\( \therefore جـ^٤ + ٤جـ^٢ د^٢ + ٤د^٤ + ٣جـ^٢ د^٢ - ٤جـ^٢ د^٢ \)
\( = (جـ^٢ + ٢د^٢)^٢ - جـ^٢ د^٢ \)
\( = [(جـ^٢ + ٢د^٢) + جـ د] [(جـ^٢ + ٢د^٢) - جـ د] \)
\( = (جـ^٢ + جـ د + ٢د^٢) (جـ^٢ - جـ د + ٢د^٢) \)
