[٢٤] \( \frac{٣س - ٨}{س - ٢} = \frac{ص - ٨}{س + ٥} \) ، بضرب الوسطين في الطرفين
\( (٣س - ٨)(س + ٥) = (س - ٢)(ص - ٨) \)
\( ٣س^٢ + ١٥س - ٨س - ٤٠ = س^٢ - ٨س - ٢س + ١٦ \)
نحصل على المعادلة \( ٢س^٢ - ٢٥س + ٥٦ = \cdot \)
حيث \( أ = ٢ ، ب = -٢٥ ، ج = ٥٦ \)
\( \Delta = (-٢٥)^٢ - ٤ \times ٢ \times ٥٦ = ٦٢٥ - ٤٤٨ = ١٧٧ > \cdot \)
حيث \( \sqrt{\Delta} = \sqrt{١٧٧} \approx ١٣,٣ \) ومنها
\( س = \frac{٢٥ \pm ١٣,٣٠}{٢ \times ٢} = \frac{٢٥ \pm ١٣,٣٠}{٤} \)
أما \( س = \frac{٢٥ + ١٣,٣٠}{٤} = \frac{٣٨,٣٠}{٤} \approx ٩,٥٧٥ \)
أو \( س = \frac{٢٥ - ١٣,٣٠}{٤} = \frac{١١,٧}{٤} \approx ٢,٩٢٥ \)
التحقق: عندما \( س = \frac{٣٨,٣٠}{٤} \) يكون الأيمن = الأيسر \( \approx ٢,٧ \)
وعندما \( س = \frac{١١,٧}{٤} \) فإن الأيمن = الأيسر \( \approx ٠,٨٤ \)
أي أن مجموعة الحل = \( \{ \frac{٣٨,٣٠}{٤} ، \frac{١١,٧}{٤} \} \)
[٢٥] \( س - \frac{٥}{س} = ٢ \) ، نتخلص من الكسر بضرب المعادلة في \( س \)
\( س^٢ - ٥ = ٢س \) ومنها \( س^٢ - ٢س - ٥ = \cdot \)
\( أ = ١ ، ب = -٢ ، ج = -٥ \) (أو نكمل المربع)
\( \Delta = (-٢)^٢ - ٤ \times ١ \times (-٥) = ٤ + ٢٠ = ٢٤ > \cdot \)
\( \therefore \sqrt{\Delta} = \sqrt{٢٤} = \sqrt{٤ \times ٦} = ٢\sqrt{٦} \)
\( س = \frac{-(-٢) \pm ٢\sqrt{٦}}{٢ \times ١} = \frac{٢ \pm ٢\sqrt{٦}}{٢} = \frac{٢(١ \pm \sqrt{٦})}{٢} \)
أما \( س = \frac{٢ + ٢\sqrt{٦}}{٢} = ١ + \sqrt{٦} \)
أو \( س = \frac{٢ - ٢\sqrt{٦}}{٢} = ١ - \sqrt{٦} \)
مجموعة الحل = \( \{ ١ + \sqrt{٦} ، ١ - \sqrt{٦} \} \)
التحقق: عندما \( س = ١ + \sqrt{٦} \) فمن \( س^٢ - ٢س - ٥ = \cdot \) نجد :
\( (١ + \sqrt{٦})^٢ - ٢(١ + \sqrt{٦}) - ٥ = (١ + ٢\sqrt{٦} + ٦) - ٢ - ٢\sqrt{٦} - ٥ \)
\( = ١ + ٦ - ٢ - ٥ = \cdot \) يكون الأيمن = الأيسر
بالمثل عندما \( س = ١ - \sqrt{٦} \) يكون الأيمن = الأيسر = صفر
[٢٦] \( \frac{س^٢ - ٤}{٢} - \frac{٢س - ٣}{٢} = \cdot \) بضرب طرفي المعادلة في ٤
\( ٢س^٢ - ٨ - ٢س + ٣ = \cdot \) ومنها \( ٢س^٢ - ٢س - ٥ = \cdot \)
يمكن حلها بإكمال المربع وبإضافة مربع نصف معامل س للطرفين
\( ٢س^٢ - ٢س = ٥ \) بالقسمة على ٢
\( س^٢ - س = \frac{٥}{٢} \) بإضافة \( (\frac{١}{٢} \times -١)^٢ = (- \frac{١}{٢})^٢ = \frac{١}{٤} \)
\( (س^٢ - س + \frac{١}{٤}) = \frac{٥}{٢} + \frac{١}{٤} \) ومنها \( (س - \frac{١}{٢})^٢ = \frac{١١}{٤} \)
بأخذ الجذر التربيعي للطرفين
\( س - \frac{١}{٢} = \pm \frac{\sqrt{١١}}{٢} \) حيث \( \sqrt{١١} \approx ٣,٣ \)
أما \( س - \frac{١}{٢} = \frac{\sqrt{١١}}{٢} \) ومنها \( س = \frac{\sqrt{١١} + ١}{٢} = \frac{٤,٣}{٢} \approx ٢,١٥ \)
أو \( س - \frac{١}{٢} = -\frac{\sqrt{١١}}{٢} \) ومنها \( س = \frac{١ - \sqrt{١١}}{٢} = \frac{-٢,٣}{٢} \approx -١,١٥ \)
مجموعة الحل = \( \{ ٢,١٥ ، -١,١٥ \} \)
التحقق: عندما \( س = \frac{\sqrt{١١}+١}{٢} \) فمن المعادلة \( ٢س^٢ - ٢س - ٥ = \cdot \)
تأكد بنفسك تحقق تساوي الطرفين عندما \( س = \frac{\sqrt{١١}-١}{٢} \)
[٢٧] \( ٢ (س + ١)^٢ - س = ٤ \)
\( ٢ (س^٢ + ٢س + ١) - س = ٤ \)
\( ٢س^٢ + ٤س + ٢ - س = ٤ \)
\( ٢س^٢ + ٣س - ٢ = \cdot \) ومنها \( (٢س - ١)(س + ٢) = \cdot \)
أما \( ٢س - ١ = \cdot \) ومنها \( س = \frac{١}{٢} \) أو \( س + ٢ = \cdot \) ومنها \( س = -٢ \)
مجموعة الحل = \( \{ -٢ ، \frac{١}{٢} \} \)
التحقق: عندما \( س = -٢ \)
الأيمن = \( ٢(-٢+١)^٢ - (-٢) = ٢(١) + ٢ = ٤ \) = الأيسر
عندما \( س = \frac{١}{٢} \)
الأيمن = \( ٢(\frac{١}{٢} + ١)^٢ - \frac{١}{٢} = ٢(\frac{٣}{٢})^٢ - \frac{١}{٢} = ٢ \times \frac{٩}{٤} - \frac{١}{٢} \)
\( = \frac{٩}{٢} - \frac{١}{٢} = \frac{٨}{٢} = ٤ \) = الأيسر
[٢٨] \( ٥س - ٧ = ٣س^٢ - س - ٢ \) نحولها إلى الصورة العامة
\( ٣س^٢ - ٦س + ٥ = \cdot \) \( أ = ٣ ، ب = -٦ ، ج = ٥ \)
\( \Delta = (-٦)^٢ - ٤ \times ٣ \times ٥ = ٣٦ - ٦٠ = -٢٤ < \cdot \)
المعادلة غير قابلة للحل في ح أي مجموعة الحل = \( \phi \)
[٢٩] \( \frac{٥س}{س - ١} = \frac{٧س}{س + ٥} \) نضرب الطرفين في الوسطين
\( ٥س(س + ٥) = ٧س(س - ١) \) نحولها إلى الصورة العامة
\( ٧س^٢ - ٧س - ٥س^٢ - ٢٥س = \cdot \) ومنها \( ٢س^٢ - ٣٢س = \cdot \) \( س(س - ١٦) = \cdot \)
أما \( ٢س = \cdot \) ومنها \( س = \cdot \)
أو \( س - ١٦ = \cdot \) ومنها \( س = ١٦ \)
مجموعة الحل = \( \{ \cdot ، ١٦ \} \)

وعندما \( س = ١٦ \)
الأيمن = \( ٧ \times ١٦ - ٣٢ \times ١٦ = \cdot \) = الأيسر

\( س(س - ٥) = ١٥٠ \)
\( س^٢ - ٥س - ١٥٠ = \cdot \) ومنها \( (س - ١٥)(س + ١٠) = \cdot \)
