تحليل المقادير الجبرية

٦س٤ − ٢٤س٢ص٢ = ٦س٢( س٢ − ٤ص٢ )

= ٦س٢( س − ٢ص )( س + ٢ص )

٢س٢ − ٤س ص = ٢س( س − ٢ص )

∴ م . م . أ للمقادير = ٦س٢( س − ٢ص )٢( س + ٢ص )

تمارين ومسائل

أوجد المضاعف المشترك الأصغر لكل مما يأتي :

[١] م ب٢ ، ب ج٢ ، أ ب٢ ج .

[٢] ٩س ص ، ١٨س٢ص ع ، ٢٧ع٢ .

[٣] ب − ب٣ ، ١ − ب٢ ، ب − ٢ .

[٤] س٣ − ٨ ، ٤س٢ + ٤س − ٨ ، س٢ − ٢س .

[٥] ٢٧س٣ − ٨ ، ٦س٣ − ١٣س٢ + ٦س ، ٤س٢ − ٩ .

[٦] ٢س٣ + ٣س ص + ص٢ ، س٢ − ص٢ .

[٧] ٤س٢ − س ، ٢س٢ − س ، س٢ − ٢ .

[٨] ٦س٢ + ٢س − ٤ ، ٥٤س٣ − ١٦ ، ١ + ٣س .

[٩] س٤ − ١٠س٢ + ٢٥ ، ٢٥س٤ + ٤س − ١ ، ١ − س .

[١٠] ٢٤س٤ص − ٨١س٢ص٤ ، ١٢س٣ + ٢٧س٢ص − ٣٦س٢ص٢ ، ٦٠س٣ص٢ − ١٣٥س ص٤ − ٩٠س٢ص٣ .
