عدد المماسات المشتركة = ٤
دائرتان متماستان من الخارج
عدد المماسات المشتركة = ٣
دائرتان متماستان من الداخل
عدد المماسات المشتركة = ١
دائرتان متقاطعتان
عدد المماسات المشتركة = ٢
ملاحظة: ١. الدائرتان المتداخلتان ( المنفصلتان من الداخل ) عدد مماساتها المشتركة = ٠.
٢. الدائرتان المنطبقتان عدد مماساتها المشتركة عدد لا نهائي.
ملخص للقواعد العامة للعلاقة بين دائرتين
نوع العلاقة بين دائرتين القاعدة
١. المتباعدتان (منفصلتان). ١. | م١ م٢ | > نق١ + نق٢.
٢. المتداخلتان (منفصلتان). ٢. | م١ م٢ | < | نق١ - نق٢ |.
٣. متحدتا المركز ( مشتركي المركز ) بحيث نق١ ≠ نق٢ (منفصلتان). ٣. | م١ م٢ | = ٠.
٤. المتماستان من الداخل. ٤. | م١ م٢ | = | نق١ - نق٢ |.
٥. المتماستان من الخارج. ٥. | م١ م٢ | = نق١ + نق٢.
٦. المتقاطعتان في نقطتين. ٦. | نق١ - نق٢ | < | م١ م٢ | < نق١ + نق٢.
٧. المتقاطعتان في أكثر من نقطتين ( منطبقتان ) بحيث نق١ = نق٢. ٧. | م١ م٢ | = ٠.
تدريب ١: اكتب الرقم من العمود (ب) في المكان المناسب من العمود (أ)، حيث نق١ = ٣ سم، نق٢ = ٥ سم:
(أ)
٨. إذا كان | م١ م٢ | = ٨ سم، فإن الدائرتين م١، م٢

١١. إذا كان | م١ م٢ | = ١ سم، فإن الدائرتين م١، م٢
١٢. إذا كان | م١ م٢ | = ٦ سم، فإن الدائرتين م١، م٢
١٣. إذا كان | م١ م٢ | = صفر سم، فإن الدائرتين م١، م٢
